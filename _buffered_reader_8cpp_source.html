32 void BufferedReader::load(const char * filename) {
33 release();
34
35 FILE *file = fopen(filename, "r");
36 if (file == NULL) {
37 throw Exception("File not found or not accessible.");
38 }
39
40 // obtain file size:
41 fseek(file, 0l, SEEK_END);
42 length = ftell(file);
43 rewind(file);
44
45 // allocate memory to contain the whole file:
46 buffer = (char*) malloc(sizeof (char)*(length + 1));
47 if (buffer != NULL) {
48 long result = fread(buffer, 1, length, file);
49 buffer[result] = 0;
50 } else {
51 throw Exception("Memory allocation error in BufferedReader.");
52
53 }
54 fclose(file);
55 }
