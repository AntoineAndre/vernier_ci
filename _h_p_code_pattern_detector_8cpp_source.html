31 void HPCodePatternDetector::takeSnapshot(int x, int y, cv::Mat image) {
32 snapshot.setConstant(0);
33 if (image.channels() > 1) {
34 cv::cvtColor(image, image, cv::COLOR_BGR2GRAY);
35 }
36
37 cv::Mat snap64f;
38 image.convertTo(snap64f, CV_64FC1);
39 snap64f /= 256;
40 int radius = snapshotSize / 2;
41 for (int col = -radius; col < radius; col++) {
42 for (int row = -radius; row < radius; row++) {
43 double distanceToCenter = sqrt((row + 0.5) * (row + 0.5) + (col + 0.5) * (col + 0.5)); // +0.5 A VERIFIER MAIS PAS CRITIQUE
44 double hanning = (1 + cos(PI * distanceToCenter / radius)) / 2;
45 if (distanceToCenter < radius && y + row > 0 && x + row > 0 && y + row < snap64f.rows && x + row < snap64f.cols) {
46 snapshot.real()(radius + row, radius + col) = hanning * snap64f.at<double>(y + row, x + col);
47 }
48 }
49 }

52 void HPCodePatternDetector::compute(const cv::Mat& image) {
53
54 detector.compute(image);
55
56 for (int i = 0; i < detector.codes.size(); i++) {
57
58 QRCode code = detector.codes[i];
59
60 if ((int) code.getRadius()*2 > snapshotSize) {
61 throw Exception("The HPCode is too large for pose estimation: increase the snapshot size.");
62 }
63 if ((int) code.getRadius() < numberHalfPeriods) {
64 throw Exception("The HPCode is too tiny for pose estimation: increase the picture quality size.");
65 }
66
67 int centerX = (int) code.center.x;
68 int centerY = (int) code.center.y;
69 takeSnapshot(centerX, centerY, image);
70 patternPhase.compute(snapshot);
71
72 double alpha;
73 double dx, dy;
74 double diffAngle = angleInPiPi(patternPhase.getPlane1().getAngle() - code.getAngle());
75 if (diffAngle >= -PI / 4 && diffAngle <= PI / 4) {
76 alpha = patternPhase.getPlane1().getAngle();
77 dx = -patternPhase.getPlane1().getPosition(physicalPeriod, 0.0, 0.0);
78 dy = -patternPhase.getPlane2().getPosition(physicalPeriod, 0.0, 0.0);
79 } else if (diffAngle >= 3 * PI / 4 || diffAngle <= -3 * PI / 4) {
80 alpha = patternPhase.getPlane1().getAngle() + PI;
81 dx = +patternPhase.getPlane1().getPosition(physicalPeriod, 0.0, 0.0);
82 dy = +patternPhase.getPlane2().getPosition(physicalPeriod, 0.0, 0.0);
83 } else if (diffAngle >= PI / 4 && diffAngle <= 3 * PI / 4) {
84 alpha = patternPhase.getPlane1().getAngle() - PI / 2;
85 dx = patternPhase.getPlane2().getPosition(physicalPeriod, 0.0, 0.0);
86 dy = -patternPhase.getPlane1().getPosition(physicalPeriod, 0.0, 0.0);
87 } else {
88 alpha = patternPhase.getPlane1().getAngle() + PI / 2;
89 dx = -patternPhase.getPlane2().getPosition(physicalPeriod, 0.0, 0.0);
90 dy = patternPhase.getPlane1().getPosition(physicalPeriod, 0.0, 0.0);
91 }
92
93 double pixelSize = physicalPeriod / patternPhase.getPixelPeriod();
94 double xImg = (centerX - image.cols / 2);
95 double yImg = (centerY - image.rows / 2);
96 double x = pixelSize * (xImg * cos(alpha) - yImg * sin(-alpha)) + dx;
97 double y = pixelSize * (xImg * sin(-alpha) + yImg * cos(alpha)) + dy;
98
99 Pose pose = Pose(x, y, alpha, pixelSize);
100
101 if ((numberHalfPeriods - 1) % 4 == 0) {
102 unsigned long number = readNumber(code, image, patternPhase.getPixelPeriod() / 2.0);
103
104 codes.insert(std::make_pair(number, pose));
105 } else {
106 codes.insert(std::make_pair(i, pose));
107 }
108
109 }
110 }

112 unsigned long HPCodePatternDetector::readNumber(QRCode& code, const cv::Mat& image, double dotSize) {
113 cv::Point2d rightDirection = (code.right - code.top);
114 rightDirection *= dotSize / cv::norm(rightDirection);
115 cv::Point2d upDirection = (code.top - code.bottom);
116 upDirection *= dotSize / cv::norm(upDirection);
117 cv::Point2d end = code.top + 3 * upDirection + 3.5 * rightDirection;
118 cv::Point2d start = code.right + 3 * upDirection - 3.5 * rightDirection;
119
120 // Getting the line
121 cv::LineIterator it(image, cv::Point2i(start), cv::Point2i(end), 8);
122 cv::Mat rasterLine(1, it.count, CV_8U);
123 for (int i = 0; i < it.count; i++, ++it) {
124 rasterLine.at<unsigned char>(0, i) = **it;
125 }
126
127 // Analysing the line
128 int dotCount = 7 + (int) std::round((cv::norm(code.top - code.right) / dotSize));
129 //std::cout << "dotCount : " << 7 + (double)(cv::norm(code.top - code.right) / dotSize) << std::endl;
130 cv::Mat dataLine(1, dotCount - 14, CV_8U);
131 cv::resize(rasterLine, dataLine, dataLine.size(), 0, 0, cv::INTER_AREA);
132 cv::threshold(dataLine, dataLine, 0, 255, cv::THRESH_BINARY | cv::THRESH_OTSU);
133 //std::cout << "data line : " << dataLine << std::endl;
134 int zeroCount = 0;
135 for (int col = 0; col < dataLine.cols; col += 2) {
136 if (dataLine.at<unsigned char>(0, col) == 0) {
137 zeroCount++;
138 }
139 }
140 int lowLevel = 0;
141 if (zeroCount < dataLine.cols / 4) {
142 lowLevel = 255;
143 }
144
145 // Reading the data line
146 unsigned long number = 0;
147 int n = 0;
148 for (int col = 1; col < dataLine.cols - 1; col += 2) {
149 if (dataLine.at<unsigned char>(0, col) == lowLevel) {
150 number |= 1 << n;
151 }
152 n++;
153 }
154
155 return number;
156 }
