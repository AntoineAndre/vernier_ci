21 void RegressionPlane::resize(int rows, int cols) {
22 if (rows <= 0 || cols <= 0) {
23 throw Exception("Can't resize a RegressionPlane with rows<=0 or cols<=0");
24 } else if (getNRows() != rows || getNCols() != cols) {
25
26 colOffset = (int) (cols * cropFactor / 2.0);
27 rowOffset = (int) (rows * cropFactor / 2.0);
28 cols -= 2 * colOffset;
29 rows -= 2 * rowOffset;
30
31 phaseCropped.resize(rows, cols);
32
33 meshRow = Eigen::VectorXd::LinSpaced(rows, -rows / 2, rows / 2 - 1).replicate(1, cols);
34 meshCol = Eigen::RowVectorXd::LinSpaced(cols, -cols / 2, cols / 2 - 1).replicate(rows, 1);
35
36 matMean << meshCol.cwiseProduct(meshCol).mean(), meshCol.cwiseProduct(meshRow).mean(), meshCol.mean(),
37 meshCol.cwiseProduct(meshRow).mean(), meshRow.cwiseProduct(meshRow).mean(), meshRow.mean(),
38 meshCol.mean(), meshRow.mean(), 1;
39 }
40 }

42 PhasePlane RegressionPlane::compute(const Eigen::ArrayXXd & unwrappedPhase) {
43 resize(unwrappedPhase.rows(), unwrappedPhase.cols());
44 Eigen::Vector3d planeCoefficients;
45 Eigen::Vector3d vecMean;
46
47 phaseCropped = unwrappedPhase.block(rowOffset, colOffset, unwrappedPhase.rows() - 2 * rowOffset, unwrappedPhase.cols() - 2 * colOffset);
48
49 vecMean.x() = meshCol.cwiseProduct(phaseCropped).mean();
50 vecMean.y() = meshRow.cwiseProduct(phaseCropped).mean();
51 vecMean.z() = phaseCropped.mean();
52
53 planeCoefficients = matMean.inverse() * vecMean;
54 return PhasePlane(planeCoefficients);
55 }

57 PhasePlane RegressionPlane::computeWithMask(const Eigen::ArrayXXd & unwrappedPhase, const Eigen::ArrayXXd & mask) {
58 resize(unwrappedPhase.rows(), unwrappedPhase.cols());
59 Eigen::Vector3d planeCoefficients;
60 Eigen::Vector3d vecMean;
61
62 phaseCropped = unwrappedPhase.block(rowOffset, colOffset, unwrappedPhase.rows() - 2 * rowOffset, unwrappedPhase.cols() - 2 * colOffset);
63 Eigen::ArrayXXd maskCropped = mask.block(rowOffset, colOffset, unwrappedPhase.rows() - 2 * rowOffset, unwrappedPhase.cols() - 2 * colOffset);
64
65 Eigen::ArrayXXd meshColMasked = meshCol.cwiseProduct(maskCropped);
66 Eigen::ArrayXXd meshRowMasked = meshRow.cwiseProduct(maskCropped);
67
68 vecMean.x() = meshColMasked.cwiseProduct(phaseCropped).mean();
69 vecMean.y() = meshRowMasked.cwiseProduct(phaseCropped).mean();
70 vecMean.z() = phaseCropped.mean();
71
72 Eigen::Matrix3d matMeanMasked;
73 matMeanMasked << meshColMasked.cwiseProduct(meshColMasked).mean(), meshColMasked.cwiseProduct(meshRowMasked).mean(), meshColMasked.mean(),
74 meshColMasked.cwiseProduct(meshRowMasked).mean(), meshRowMasked.cwiseProduct(meshRowMasked).mean(), meshRowMasked.mean(),
75 meshColMasked.mean(), meshRowMasked.mean(), 1;
76
77 planeCoefficients = matMeanMasked.inverse() * vecMean;
78 return PhasePlane(planeCoefficients);
79 }
