63 double PhasePlane::getPosition(double physicalPeriod, double y, double x, int periodShift) {
64 double phiCenter = getPhase(y, x);
65 if (physicalPeriod != 0.0) {
66 return physicalPeriod * (phiCenter / (2.0 * PI) + (double) (periodShift));
67 } else {
68 return (2.0 * PI / (std::sqrt(pow(a, 2) + pow(b, 2)))) * (phiCenter / (2.0 * PI) + (double) (periodShift));
69 }
70 }

72 double PhasePlane::getPositionPixels(double y, double x, int periodShift) {
73 double phiCenter = getPhase(y, x);
74 return (1.0 / std::sqrt(a * a + b * b)) * (phiCenter + (double) (periodShift));
75 }
