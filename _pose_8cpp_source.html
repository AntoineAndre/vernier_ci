46 cos(beta) * sin(alpha), cos(alpha) * cos(gamma) + sin(alpha) * sin(beta) * sin(gamma), cos(gamma) * sin(alpha) * sin(beta) - cos(alpha) * sin(gamma), y * (cos(alpha) * cos(gamma) + sin(alpha) * sin(beta) * sin(gamma)) + x * cos(beta) * sin(alpha),
47 -sin(beta), cos(beta) * sin(gamma), cos(beta) * cos(gamma), z - x * sin(beta) + y * cos(beta) * sin(gamma),

76 void Pose::draw(cv::Mat & image, double length, std::string name) {
77 if (length < 0.0) {
78 length = image.cols / 4;
79 }
80 Eigen::Matrix4d cTp = getCameraToPatternTransformationMatrix();
81 double xImg = cTp(0, 3) / pixelSize + image.cols / 2;
82 double yImg = cTp(1, 3) / pixelSize + image.rows / 2;
83 int thickness = 2;
84 cv::line(image, cv::Point(xImg, yImg), cv::Point(xImg + length * cos(alpha), yImg + length * sin(alpha)), cv::Scalar(0, 0, 255), thickness, cv::LINE_AA);
85 cv::line(image, cv::Point(xImg, yImg), cv::Point(xImg + length * cos(alpha + PI / 2), yImg + length * sin(alpha + PI / 2)), cv::Scalar(0, 255, 0), thickness, cv::LINE_AA);
86 cv::line(image, cv::Point(xImg, yImg), cv::Point(xImg - length * cos(alpha), yImg - length * sin(alpha)), cv::Scalar(255, 150, 128), thickness, cv::LINE_AA);
87 cv::line(image, cv::Point(xImg, yImg), cv::Point(xImg - length * cos(alpha + PI / 2), yImg - length * sin(alpha + PI / 2)), cv::Scalar(255, 150, 128), thickness, cv::LINE_AA);
88 cv::putText(image, name, cv::Point(xImg + 5, yImg - 5), cv::FONT_HERSHEY_COMPLEX, 1, cv::Scalar(0, 0, 255), 1);
