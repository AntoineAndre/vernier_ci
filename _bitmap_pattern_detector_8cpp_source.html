85 void BitmapPatternDetector::computeThumbnail(const Eigen::ArrayXXd& array, double deltaPhase) {
86 double approxPixelPeriod = (plane1.getPixelicPeriod() + plane2.getPixelicPeriod()) / 2.0;
87 int length1 = (int) (2.82 * array.rows() / approxPixelPeriod);
88 int length2 = (int) (2.82 * array.cols() / approxPixelPeriod);
89 if (length1 % 2 == 0) length1++;
90 if (length2 % 2 == 0) length2++;
91
92 thumbnail = cv::Mat(length2, length1, CV_8U);
93
94 Eigen::ArrayXXd numberWhiteDots(length1, length2);
95 Eigen::ArrayXXd cumulWhiteDots(length1, length2);
96 Eigen::ArrayXXd numberBackgroundDots(length1, length2);
97 Eigen::ArrayXXd cumulBackgroundDots(length1, length2);
98
99 numberWhiteDots.fill(0);
100 cumulWhiteDots.fill(0);
101 numberBackgroundDots.fill(0);
102 cumulBackgroundDots.fill(0);
103
104 int phaseIteration1, phaseIteration2;
105
106 Eigen::Vector3d tempPlane1Coeff(plane1.getA(), plane1.getB(), plane1.getC());
107 Eigen::Vector3d tempPlane2Coeff(plane2.getA(), plane2.getB(), plane2.getC());
108
109 PhasePlane tempPlane1(tempPlane1Coeff);
110 PhasePlane tempPlane2(tempPlane2Coeff);
111
112
113 for (int col = 0; col < array.cols(); col++) {
114 for (int row = 0; row < array.rows(); row++) {
115
116 double phaseCol = tempPlane1.getPhase(row - array.rows() / 2, col - array.cols() / 2);
117 double phaseRow = tempPlane2.getPhase(row - array.rows() / 2, col - array.cols() / 2);
118
119 phaseIteration1 = round(phaseCol / PI) + length1 / 2;
120 phaseIteration2 = round(phaseRow / PI) + length2 / 2;
121
122 if (phaseIteration1 < cumulBackgroundDots.rows() && phaseIteration2 < cumulBackgroundDots.cols() && phaseIteration1 >= 0 && phaseIteration2 >= 0) {
123 if ((abs(std::fmod(phaseCol, PI)) <= deltaPhase || abs(std::fmod(phaseCol, PI)) >= PI - deltaPhase) && (abs(std::fmod(phaseRow, PI)) <= deltaPhase || abs(std::fmod(phaseRow, PI)) >= PI - deltaPhase)) {
124 numberWhiteDots(phaseIteration1, phaseIteration2) += 1;
125 cumulWhiteDots(phaseIteration1, phaseIteration2) += array(row, col);
126 }
127 }
128 }
129 }
130
131 for (int col = 0; col < thumbnail.cols; col++) {
132 for (int row = 0; row < thumbnail.rows; row++) {
133 thumbnail.at<char>(row, col) = (char) (255 * cumulWhiteDots(col, row) / numberWhiteDots(col, row));
134 }
135 }
136 }
