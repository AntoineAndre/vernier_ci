17 void MegarenaCell::getGlobalCell(Eigen::ArrayXXd thumbnailNumberWhiteDots, Eigen::ArrayXXd thumbnailCumulWhiteDots) {
18 Eigen::Array33d sumOnlyDotsRemain = Eigen::Array33d::Zero(3, 3);
19 Eigen::Array33d countOnlyDots = Eigen::Array33d::Zero(3, 3);
20
21 for (int col = 0; col < thumbnailNumberWhiteDots.cols(); col++) {
22 for (int row = 0; row < thumbnailNumberWhiteDots.rows(); row++) {
23 int index1 = row % 3;
24 int index2 = col % 3;
25
26 sumOnlyDotsRemain(index1, index2) += thumbnailCumulWhiteDots(row, col);
27 countOnlyDots(index1, index2) += thumbnailNumberWhiteDots(row, col);
28 }
29 }
30
31 sumOnlyDotsRemain = sumOnlyDotsRemain / countOnlyDots;
32
33 this->globalCell = sumOnlyDotsRemain;
34 }
