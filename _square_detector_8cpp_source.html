71 void SquareDetector::compute(const cv::Mat& image) {
72
73 cv::aruco::ArucoDetector detector(cv::aruco::getPredefinedDictionary(cv::aruco::DICT_6X6_250), parameters);
74
75 std::vector<std::vector<cv::Point> > contours;
76 std::vector<std::vector<cv::Point2f> > candidates;
77 detector.arucoDetectorImpl->detectCandidates(image, candidates, contours);
78
79 std::vector<cv::aruco::MarkerCandidateTree> selectedCandidates;
80 selectedCandidates = detector.arucoDetectorImpl->filterTooCloseCandidates(candidates, contours);
81
82 squares.clear();
83 for (int i = 0; i < selectedCandidates.size(); i++) {
84 if (selectedCandidates[i].parent < 0) {
85 Square s(selectedCandidates[i].corners);
86 squares.push_back(s);
87 }
88 }
