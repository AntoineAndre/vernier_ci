20 void PeriodicPatternLayout::resize(double period, int nRows, int nCols) {
21 if (nRows <= 0 || nRows % 2 == 0) {
22 throw Exception("The number of rows must be positive and odd.");
23 }
24 if (nCols <= 0 || nCols % 2 == 0) {
25 throw Exception("The number of columns must be positive and odd.");
26 }
27 this->period = period;
28 this->dotSize = 0.5 * period;
29 this->nRows = nRows;
30 this->nCols = nCols;
31 width = period * (nCols - 0.5);
32 height = period * (nRows - 0.5);
33 originX = 0.5 * width;
34 originY = 0.5 * height;
35 }

44 void PeriodicPatternLayout::readJSON(const rapidjson::Value & document) {
45
46 PatternLayout::readJSON(document);
47
48 if (document.HasMember("period") && document["period"].IsDouble()) {
49 period = document["period"].GetDouble();
50 } else {
51 throw Exception("The file is not a valid periodic pattern file, the period is missing or has a wrong format.");
52 }
53 if (document.HasMember("nRows") && document["nRows"].IsInt()) {
54 nRows = document["nRows"].GetInt();
55 } else {
56 throw Exception("The file is not a valid periodic pattern file, the number of rows (nRows) is missing or has a wrong format.");
57 }
58 if (document.HasMember("nCols") && document["nCols"].IsInt()) {
59 nCols = document["nCols"].GetInt();
60 } else {
61 throw Exception("The file is not a valid periodic pattern file, the number of columns (nCols) is missing or has a wrong format.");
62 }
63 resize(period, nRows, nCols);
64
65 if (document.HasMember("dotSize") && document["dotSize"].IsDouble()) {
66 dotSize = document["dotSize"].GetDouble();
67 }

70 void PeriodicPatternLayout::toRectangleVector(std::vector<Rectangle>& rectangleList) {
71 double offset = (period / 2 - dotSize) / 2;
72 for (int col = 0; col < nCols; col++) {
73 double x = col * period + offset;
74 for (int row = 0; row < nRows; row++) {
75 double y = row * period + offset;
76 if (row != 0 || col != 0) {
77 rectangleList.push_back(Rectangle(x, y, dotSize, dotSize));
78 }
79 }
80 }

83 double PeriodicPatternLayout::getIntensity(double x, double y) {
84 if (x < -0.5 * width || y < -0.5 * height || x > 0.5 * width || y > 0.5 * height) {
85 return 0;
86 } else if (x < -0.5 * width + period && y < -0.5 * height + period) {
87 return 0;
88 } else {
89 return (1 + cos(2 * PI * x / period)) * (1 + cos(2 * PI * y / period)) / 4;
90 }
91 }

93 void PeriodicPatternLayout::saveToPNG(std::string filename) {
94 cv::Mat image(2 * nRows - 1, 2 * nCols - 1, CV_8U);
95 for (int col = 0; col < image.cols; col++) {
96 double x = col * period * 0.5 + 0.25 * period - originX;
97 for (int row = 0; row < image.rows; row++) {
98 double y = row * period * 0.5 + 0.25 * period - originY;
99 image.at<char>(row, col) = (char) (255 * (getIntensity(x, y) > 0.5));
100 }
101 }
102 if (filename == "") {
103 filename = classname + ".png";
104 }
105 cv::imwrite(filename, image);
106 }
