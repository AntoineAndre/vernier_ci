31 void StampPatternDetector::takeSnapshot(int x, int y, cv::Mat image) {
32 snapshot.setConstant(0);
33 if (image.channels() > 1) {
34 cv::cvtColor(image, image, cv::COLOR_BGR2GRAY);
35 }
36
37 cv::Mat snap64f;
38 image.convertTo(snap64f, CV_64FC1);
39 snap64f /= 256;
40 int radius = snapshotSize / 2;
41 for (int col = -radius; col < radius; col++) {
42 for (int row = -radius; row < radius; row++) {
43 double distanceToCenter = sqrt((row + 0.5) * (row + 0.5) + (col + 0.5) * (col + 0.5)); // +0.5 A VERIFIER MAIS PAS CRITIQUE
44 double hanning = (1 + cos(PI * distanceToCenter / radius)) / 2;
45 if (distanceToCenter < radius && y + row > 0 && x + row > 0 && y + row < snap64f.rows && x + row < snap64f.cols) {
46 snapshot.real()(radius + row, radius + col) = hanning * snap64f.at<double>(y + row, x + col);
47 }
48 }
49 }

52 void StampPatternDetector::compute(const cv::Mat& image) {
53 cv::Mat grayImage;
54 if (image.channels() > 1) {
55 cv::cvtColor(image, grayImage, cv::COLOR_BGR2GRAY);
56 } else {
57 grayImage = image;
58 }
59 //cv::normalize(grayImage, grayImage, 255, 0, cv::NORM_MINMAX);
60 grayImage.convertTo(grayImage, CV_8UC1);
61
62 detector.compute(image);
63
64 stamps.clear();
65 for (int i = 0; i < detector.squares.size(); i++) {
66
67 Square square = detector.squares[i];
68
69 if ((int) square.getRadius() > snapshotSize) {
70 throw Exception("The QRCode is too large for pose estimation: increase the snapshot size.");
71 }
72 if ((int) square.getRadius() < numberHalfPeriods) {
73 throw Exception("The QRCode is too tiny for pose estimation: increase the picture quality size.");
74 }
75
76 int centerX = (int) square.getCenter().x;
77 int centerY = (int) square.getCenter().y;
78 takeSnapshot(centerX, centerY, image);
79 patternPhase.compute(snapshot);
80
81 double alpha;
82 double dx, dy;
83 double diffAngle = angleInPiPi(patternPhase.getPlane1().getAngle() - square.getAngle());
84 if (diffAngle >= -PI / 4 && diffAngle <= PI / 4) {
85 alpha = patternPhase.getPlane1().getAngle();
86 dx = -patternPhase.getPlane1().getPosition(physicalPeriod, 0.0, 0.0);
87 dy = -patternPhase.getPlane2().getPosition(physicalPeriod, 0.0, 0.0);
88 } else if (diffAngle >= 3 * PI / 4 || diffAngle <= -3 * PI / 4) {
89 alpha = patternPhase.getPlane1().getAngle() + PI;
90 dx = +patternPhase.getPlane1().getPosition(physicalPeriod, 0.0, 0.0);
91 dy = +patternPhase.getPlane2().getPosition(physicalPeriod, 0.0, 0.0);
92 } else if (diffAngle >= PI / 4 && diffAngle <= 3 * PI / 4) {
93 alpha = patternPhase.getPlane1().getAngle() - PI / 2;
94 dx = patternPhase.getPlane2().getPosition(physicalPeriod, 0.0, 0.0);
95 dy = -patternPhase.getPlane1().getPosition(physicalPeriod, 0.0, 0.0);
96 } else {
97 alpha = patternPhase.getPlane1().getAngle() + PI / 2;
98 dx = -patternPhase.getPlane2().getPosition(physicalPeriod, 0.0, 0.0);
99 dy = patternPhase.getPlane1().getPosition(physicalPeriod, 0.0, 0.0);
100 }
101
102 double pixelSize = physicalPeriod / patternPhase.getPixelPeriod();
103 double xImg = (centerX - image.cols / 2);
104 double yImg = (centerY - image.rows / 2);
105 double x = pixelSize * (xImg * cos(alpha) - yImg * sin(-alpha)) + dx;
106 double y = pixelSize * (xImg * sin(-alpha) + yImg * cos(alpha)) + dy;
107
108 Pose pose = Pose(x, y, alpha, pixelSize);
109
110 stamps.push_back(pose);
111 }
112 }
