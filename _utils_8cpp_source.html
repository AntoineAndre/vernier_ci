11 void drawCameraFrame(cv::Mat& image) {

20 cv::Mat array2image(const Eigen::ArrayXXd & array) {
21 Eigen::MatrixXd matrix;
22 matrix = (array.array()-array.minCoeff())/array.maxCoeff();
23
24 cv::Mat image64d;
25 cv::eigen2cv(matrix, image64d);
26 cv::Mat image8u(image64d.rows, image64d.cols, CV_8UC4);
27
28 for (int row = 0; row < image64d.rows; ++row) {
29 double* src = image64d.ptr<double>(row);
30 uchar* dst = image8u.ptr<uchar>(row);
31 for (int col = 0; col < image64d.cols; ++col) {
32 uchar value;
33 if (src[col] >= 1.0) {
34 value = (uchar) 255;
35 } else if (src[col] <= 0.0) {
36 value = (uchar) 0;
37 } else {
38 value = (uchar) (src[col]*255);
39 }
40 dst[4 * col] = dst[4 * col + 1] = dst[4 * col + 2] = value;
41 dst[4 * col + 3] = (uchar) 255;
42 }
43 }
44 return image8u;
45 }

53 Eigen::ArrayXXd image2array(const cv::Mat & image) {
54 cv::Mat grayImage;
55 if (image.channels() > 1) {
56 cv::cvtColor(image, grayImage, cv::COLOR_BGR2GRAY);
57 } else {
58 grayImage = image;
59 }
60
61 grayImage.convertTo(grayImage, CV_64F);
62 cv::normalize(grayImage, grayImage, 1.0, 0, cv::NORM_MINMAX);
63 Eigen::MatrixXd patternMatrix;
64 cv::cv2eigen(grayImage, patternMatrix);
65 Eigen::ArrayXXd patternArray;
66 patternArray = patternMatrix.array();
67 return patternArray;
68 }
