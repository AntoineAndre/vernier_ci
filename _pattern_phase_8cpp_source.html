49 void PatternPhase::compute(const Eigen::ArrayXXcd& patternArray) {
50 resize(patternArray.rows(), patternArray.cols());
51
52 fft.compute(patternArray, spectrum);
53
54 Spectrum::shift(spectrum, spectrumShifted);
55 spectrumFiltered1 = spectrumShifted;
56 spectrumFiltered2 = spectrumShifted;
57
58 if (pixelPeriod == 0.0) {
59 Spectrum::mainPeakHalfPlane(spectrumShifted, mainPeak1, mainPeak2);
60 } else {
61 switch (peaksSearchMethod) {
62 case 0:
63 Spectrum::mainPeakHalfPlane(spectrumShifted, mainPeak1, mainPeak2);
64 break;
65 case 1:
66 Spectrum::mainPeak4Search(spectrumShifted, mainPeak1, mainPeak2);
67 break;
68 case 2:
69 Spectrum::mainPeakPerimeter(spectrumShifted, mainPeak1, mainPeak2);
70 break;
71 default:
72 Spectrum::mainPeakCircle(spectrumShifted, mainPeak1, mainPeak2, pixelPeriod);
73 ;
74 break;
75 }
76 }
77
78 // Compute first plane phase from peak 1
79 gaussianFilter.applyTo(spectrumFiltered1, mainPeak1(1), mainPeak1(0));
80
81 ifft.compute(spectrumFiltered1, phase1);
82 Spatial::shift(phase1);
83
84 unwrappedPhase1 = phase1.arg();
85
86 Spatial::quartersUnwrapPhase(unwrappedPhase1);
87
88 plane1 = regressionPlane.compute(unwrappedPhase1);
89
90 this->pixelPeriod = plane1.getPixelicPeriod();
91
92 // Compute second plase from peak 2
93 gaussianFilter.applyTo(spectrumFiltered2, mainPeak2(1), mainPeak2(0));

95 Spatial::shift(phase2);
96

104 // plane1.setC(-plane1.getC()); // supprimé le 19/11/2022 quelle différence avec oouda fft ?

106#endif

355 double PatternPhase::computeFirst(Eigen::ArrayXXcd& patternArray, double& pixelPeriod) {
356 Eigen::ArrayXXcd meanPattern = patternArray - patternArray.mean();
357 fft.compute(meanPattern, spectrum);
358 Spectrum::shift(spectrum, spectrumShifted);
359 spectrumFiltered1 = spectrumShifted;
360 Eigen::ArrayXXcd spectrumShift2(patternArray.rows(), patternArray.cols());
361 Eigen::ArrayXXcd gaussian2D(patternArray.rows(), patternArray.cols());
362 gaussian2D.setConstant(1);
363 Eigen::ArrayXXcd Jcriterion(patternArray.rows(), patternArray.cols());
364
365 Spectrum::mainPeakQuarter(spectrumShifted, mainPeak1, mainPeak2);
366
367 //std::cout << "main peak : " << mainPeak1(1) << " ; " << mainPeak1(0) << std::endl;
368
369 double sigma = 0.333;
370 spectrumShift2 = spectrumShifted;
371
372 gaussianFilter.setSigma(sigma);
373 gaussianFilter.applyTo(gaussian2D, mainPeak1(1), mainPeak1(0));
374
375 gaussian2D *= spectrumShift2(mainPeak1(1), mainPeak1(0));
376
377 gaussian2D = gaussian2D.abs();
378 spectrumShift2 = spectrumShift2.abs();
379
380 spectrumShift2 -= gaussian2D;
381
382 Jcriterion = spectrumShift2 * spectrumShift2;
383
384 double meanJPrev = Jcriterion.abs().sum() * 10;
385
386 double meanJ = 0;
387 while (true) {
388 sigma += 0.001;
389
390 spectrumShift2 = spectrumShifted;
391 gaussian2D.setConstant(1);
392 gaussianFilter.setSigma(sigma);
393 gaussianFilter.applyTo(gaussian2D, mainPeak1(1), mainPeak1(0));
394
395 gaussian2D *= spectrumShift2(mainPeak1(1), mainPeak1(0));
396
397 gaussian2D = gaussian2D.abs();
398
399 spectrumShift2 -= gaussian2D;
400
401 Jcriterion = spectrumShift2 * spectrumShift2;
402
403 meanJ = Jcriterion.abs().sum();
404
405 if (meanJ > meanJPrev) break;
406 else {
407 meanJPrev = meanJ;
408 }
409
410 //std::cout << "mean criterion : " << meanJ << " ; sigma : " << sigma << std::endl;
411 }
412
413 gaussianFilter.applyTo(spectrumFiltered1, mainPeak1(1), mainPeak1(0));
414 ifft.compute(spectrumFiltered1, phase1);
415 Spatial::shift(phase1);
416 unwrappedPhase1 = phase1.array().arg();
417 Spatial::quartersUnwrapPhase(unwrappedPhase1);
418 // phaseCropped = phase1.block(sideOffset, sideOffset, phase1.rows() - 2 * sideOffset, phase1.cols() - 2 * sideOffset);
419 this->plane1 = regressionPlane.compute(unwrappedPhase1);
420
421 pixelPeriod = plane1.getPixelicPeriod();
422
423 return sigma;
424 }
