43 void CustomPatternLayout::readJSON(const rapidjson::Value & document) {
44 PatternLayout::readJSON(document);
45 int nRows = 0;
46 if (document.HasMember("dots") && document["dots"].IsArray()) {
47 nRows = document["dots"].Size();
48 dots.resize(nRows);
49 dotsIntensity.resize(nRows);
50 } else {
51 throw Exception("The file is not a valid custom pattern file, the array of dots is missing or has a wrong format.");
52 }
53
54 if (document["dots"][0].IsArray() && document["dots"][0].Size() == 5) {
55 ;
56 } else {
57 throw Exception("The file is not a valid custom pattern file, the first row of the array of dots has a wrong format.");
58 }
59
60 for (rapidjson::SizeType row = 0; row < nRows; row++) {
61 const rapidjson::Value& value = document["dots"][row];
62 if (value.IsArray() && value.Size() == 5) {
63 if (value[0].IsDouble() && value[1].IsDouble() && value[2].IsDouble() && value[3].IsDouble() && value[4].IsDouble()) {
64 dots[row] = Rectangle(value[0].GetDouble(), value[1].GetDouble(), value[2].GetDouble(), value[3].GetDouble());
65 dotsIntensity[row] = value[4].GetDouble();
66 } else {
67 Exception("The file is not a valid custom pattern file, the row " + to_string(row) + " of the array of dots has a wrong format");
68 }
69 } else {
70 throw Exception("The file is not a valid custom pattern file, the row " + to_string(row) + " of the array of dots has a wrong size");
71 }
72 }
73
74 resize();
75 }

77 void CustomPatternLayout::loadFromCSV(std::string filename) {
78 dots.clear();
79 dotsIntensity.clear();
80
81 std::ifstream file(filename);
82 if (!file.good()) {
83 throw Exception("File not found or not accessible.");
84 }
85
86 std::string line = "";
87 std::getline(file, line);
88
89 double x, y, w, h, intensity;
90 while (!file.eof()) {
91 std::getline(file, line);
92 if (!file.fail() && !line.empty()) {
93 std::replace(line.begin(), line.end(), ';', ' ');
94 std::stringstream stream(line);
95 stream >> x >> y >> w >> h >> intensity;
96 dots.push_back(Rectangle(x, y, w, h));
97 dotsIntensity.push_back(intensity);
98 }
99 }
100 file.close();
101
102 if (dots.size() == 0) {
103 throw Exception(filename + " is not a valid CSV file.");
104 }
105
106 resize();
107 description = "Layout created from " + filename;
108 }

114 double CustomPatternLayout::getIntensity(double x, double y) {
115 double intensity = 0.0;
116 for (int i = 0; i < dots.size(); i++) {
117 if (x >= dots[i].x - 0.5 * dots[i].width &&
118 x <= dots[i].x + 1.5 * dots[i].width &&
119 y >= dots[i].y - 0.5 * dots[i].height &&
120 y <= dots[i].y + 1.5 * dots[i].height) {
121 intensity += (1 + cos(PI * (x - (dots[i].x + 0.5 * dots[i].width)) / dots[i].width))
122 * (1 + cos(PI * (y - (dots[i].y + 0.5 * dots[i].height)) / dots[i].height)) / 4;
123 }
124 }
125 return intensity;
126 }
