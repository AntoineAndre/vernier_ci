37 void FourierTransform::resize(int nRows, int nCols, int sign) {
38 if (nRows <= 0 || nCols <= 0) {
39 throw Exception("Can't resize a FourierTransform with rows<=0 or cols<=0");
40 } else if (nRows != this->nRows || nCols != this->nCols || sign != this->sign) {
41 if (plan != NULL) {
42 fftw_destroy_plan(plan);
43 }
44
45 this->nRows = nRows;
46 this->nCols = nCols;
47 this->sign = sign;
48
49 fftw_complex* in = (fftw_complex*) fftw_malloc(sizeof (fftw_complex) * nRows * nCols);
50 fftw_complex* out = (fftw_complex*) fftw_malloc(sizeof (fftw_complex) * nRows * nCols);
51
52 if (nRows == 1 || nCols == 1) {
53 plan = fftw_plan_dft_1d(nRows * nCols, in, out, sign, FFTW_MEASURE);
54 } else {
55 plan = fftw_plan_dft_2d(nCols, nRows, in, out, sign, FFTW_MEASURE);
56 }
57
58 fftw_free(in);
59 fftw_free(out);
60 }
61 }

115 void FourierTransform::resize(int nRows, int nCols, int sign) {
116 if (nRows <= 0 || nCols <= 0) {
117 throw Exception("Can't resize a FourierTransform with rows<=0 or cols<=0");
118 } else if (((nRows & (nRows - 1)) != 0) || ((nCols & (nCols - 1)) != 0)) {
119 throw Exception("The dimensions of a FourierTransform must be a power of two");
120 } else if (nRows != this->nRows || nCols != this->nCols || sign != this->sign) {
121 if (workArea != NULL) {
122 free(data);
123 free(workArea);
124 free(bitReversal);
125 free(cosSinTable);
126 }
127
128 this->nRows = nRows;
129 this->nCols = nCols;
130 this->sign = sign;
131
132 data = (double**) malloc(sizeof (double*) * nCols);
133 workArea = NULL;
134 int n = 2 + (int) sqrt(std::max(nRows, nCols));
135 bitReversal = (int*) malloc(sizeof (int) * n);
136 n = std::max(nRows, nCols) / 2;
137 cosSinTable = (double*) malloc(sizeof (double) * n);
138 bitReversal[0] = 0;
139 }
140 }

142 void FourierTransform::compute(Eigen::ArrayXXcd& in, Eigen::ArrayXXcd& out) {
143 resize(in.rows(), in.cols(), sign);
144 out = in;
145 // Eigen is column major but Ooura's fft is row major
146 for (int j = 0; j < out.cols(); j++) {
147 data[j] = (double*) (&out(0, j));
148 }
149 cdft2d(nCols, 2 * nRows, sign, data, workArea, bitReversal, cosSinTable);
150 out = out.conjugate();
151 }
