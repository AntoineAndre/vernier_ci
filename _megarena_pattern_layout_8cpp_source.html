33 void MegarenaPatternLayout::resize(double period) {
34 if (period < 0.0) {
35 throw Exception("The period must be positive.");
36 }
37 if (bitSequence.rows() != 1) {
38 throw Exception("The bit sequence must have a single row");
39 }
40 if (bitSequence.cols() <= 0) {
41 throw Exception("The bit sequence must have at least one column.");
42 }
43 this->period = period;
44 this->dotSize = 0.5 * period;
45 this->nRows = bitSequence.cols();
46 this->nCols = bitSequence.cols();
47 width = period * nCols - 0.5 * period;
48 height = period * nRows - 0.5 * period;
49 originX = 0.25 * period;
50 originY = 0.25 * period;
51 regionOfInterest = Rectangle(0.0, 0.0, width, height);
52 }

73 if (document.HasMember("period") && document["period"].IsDouble()) {

81 codeDepth = MegarenaBitSequence::codeDepth(bitSequence.cols());
82 for (rapidjson::SizeType col = 0; col < document["bitSequence"].Size(); col++) {
83 if (document["bitSequence"][col].IsInt()) {
84 bitSequence(col) = document["bitSequence"][col].GetInt();
85 } else {
86 Exception("The file is not a valid megarena pattern file, the col " + to_string(col) + " of the bitSequence has a wrong format");
87 }
88 }
89 } else if (document.HasMember("codeDepth") && document["codeDepth"].IsInt()) {
90 codeDepth = document["codeDepth"].GetInt();
91 if (codeDepth >= 4 && codeDepth <= 12) {
92 MegarenaBitSequence::generate(codeDepth, bitSequence);
93 } else {
94 throw Exception("The file is not a valid megarena pattern file, the code depth must between 4 and 12.");
95 }
96 } else if (document.HasMember("codeSize") && document["codeSize"].IsInt()) {
97 codeDepth = document["codeSize"].GetInt();
98 if (codeDepth >= 4 && codeDepth <= 12) {
99 MegarenaBitSequence::generate(codeDepth, bitSequence);
100 } else {
101 throw Exception("The file is not a valid megarena pattern file, the code depth must between 4 and 12.");
102 }
103 } else {
104 throw Exception("The file is not a valid megarena pattern file, the code depth is missing or has a wrong format.");
105 }
106 resize(period);

108 if (document.HasMember("regionOfInterest") && document["regionOfInterest"].IsArray()) {
109 if (document["regionOfInterest"].Size() == 4 && document["regionOfInterest"][0].IsDouble() && document["regionOfInterest"][1].IsDouble() && document["regionOfInterest"][2].IsDouble() && document["regionOfInterest"][3].IsDouble()) {

124 void MegarenaPatternLayout::toRectangleVector(std::vector<Rectangle>& rectangleList) {
125 int colStart = (int) (regionOfInterest.x / period);
126 int colStop = (int) ((regionOfInterest.x + regionOfInterest.width) / period);
127 int rowStart = (int) (regionOfInterest.y / period);
128 int rowStop = (int) ((regionOfInterest.y + regionOfInterest.height) / period);
129
130 if (colStart < 0) {
131 colStart = 0;
132 }
133 if (rowStart < 0) {
134 rowStart = 0;
135 }
136 if (colStop > nCols) {
137 colStop = nCols;
138 }
139 if (rowStop > nRows) {
140 rowStop = nRows;
141 }
142 double offset = (period / 2 - dotSize) / 2;
143 for (int col = colStart; col < colStop; col++) {
144 double x = col * period + offset;
145 for (int row = rowStart; row < rowStop; row++) {
146 double y = row * period + offset;
147 if (bitSequence(row) && bitSequence(col) && (col % 3 != 0 || row % 3 != 0)) {
148 rectangleList.push_back(Rectangle(x, y, dotSize, dotSize));
149 }
150 }
151 }

155 double MegarenaPatternLayout::getIntensity(double x, double y) {
156 if (x<-0.5 * period || y<-0.5 * period || x > width || y > height) {
157 return 0;
158 } else {
159 int col = (int) (x / period + 0.5);
160 int row = (int) (y / period + 0.5);
161 if (col < 0 || row < 0 || col >= nCols || row >= nRows) {
162 return 0;
163 } else {
164 if (bitSequence(row) && bitSequence(col) && (col % 3 != 0 || row % 3 != 0)) {
165 return (1 + cos(2 * PI * x / period)) * (1 + cos(2 * PI * y / period)) / 4;
166 } else {
167 return 0;
168 }
169 }
170 }

173 void MegarenaPatternLayout::saveToPNG(std::string filename) {
174 int colStart = (int) (regionOfInterest.x / (period * 0.5));
175 int colStop = (int) ((regionOfInterest.x + regionOfInterest.width) / (period * 0.5));
176 int rowStart = (int) (regionOfInterest.y / (period * 0.5));
177 int rowStop = (int) ((regionOfInterest.y + regionOfInterest.height) / (period * 0.5));
178
179 cv::Mat image(rowStop - rowStart, colStop - colStart, CV_8U);
180 for (int col = colStart; col < colStop; col++) {
181 double x = col * period * 0.5 + 0.25 * period - originX;
182 for (int row = rowStart; row < rowStop; row++) {
183 double y = row * period * 0.5 + 0.25 * period - originY;
184 image.at<char>(row - rowStart, col - colStart) = (char) (255 * (getIntensity(x, y) > 0.5));
185 }
186 }
187 if (filename == "") {
188 filename = classname + ".png";
189 }
190 cv::imwrite(filename, image);
191 }
